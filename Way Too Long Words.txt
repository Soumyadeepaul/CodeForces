#include <bits/stdc++.h>
using namespace std;


int main(){
int n;
cin>>n;
vector<string> words;
string word;
for(int i=0;i<n;i++){
cin>>word;
words.push_back(word);
}
int size;
for(int i=0;i<n;i++){
size=words[i].size();
if(size>10){
words[i]=words[i][0]+to_string(size-2)+words[i][size-1];
}
}
for(int i=0;i<n;i++){
cout<<words[i]<<endl;
}
return 0;
}
